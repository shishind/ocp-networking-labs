# OCP Networking Labs - Required Tools and Dependencies
# This file lists all tools needed for Week 1-7 labs

# ============================================
# BASIC NETWORKING TOOLS
# ============================================
# Core IP and network utilities
iproute2          # ip, ss, bridge commands
iputils-ping      # ping command
net-tools         # ifconfig, netstat, route (legacy but useful)
bind-utils        # dig, nslookup, host (RHEL/Fedora)
dnsutils          # dig, nslookup, host (Ubuntu/Debian)
netcat            # nc command for network testing
netcat-openbsd    # Alternative netcat (Ubuntu/Debian)
nmap-ncat         # Alternative netcat (RHEL/Fedora)
tcpdump           # Packet capture and analysis
curl              # HTTP client
wget              # Download utility
socat             # Socket relay tool
traceroute        # Network path tracing
mtr               # Network diagnostic tool
ethtool           # Ethernet device configuration
iptables          # Netfilter administration
nftables          # Modern firewall (RHEL 8+/Fedora/Ubuntu 20.04+)

# ============================================
# CONTAINER RUNTIME & TOOLS
# ============================================
docker            # Docker container runtime
docker-compose    # Docker Compose for multi-container apps
podman            # Alternative container runtime (OCP uses CRI-O based on podman)
buildah           # Container image builder
skopeo            # Container image operations

# ============================================
# KUBERNETES TOOLS
# ============================================
kubectl           # Kubernetes CLI
kind              # Kubernetes in Docker (local clusters)
helm              # Kubernetes package manager (optional but useful)

# ============================================
# OVS/OVN TOOLS (Week 7)
# ============================================
openvswitch       # Open vSwitch
openvswitch-switch # OVS daemon (Ubuntu/Debian)
ovn               # OVN components
ovn-central       # OVN central components
ovn-host          # OVN host components

# ============================================
# DEVELOPMENT & UTILITIES
# ============================================
git               # Version control
jq                # JSON processor
yq                # YAML processor
vim               # Text editor
nano              # Alternative text editor
bash-completion   # Bash completion scripts
make              # Build automation
python3           # Python for scripting
python3-pip       # Python package manager

# ============================================
# SYSTEM UTILITIES
# ============================================
util-linux        # Various system utilities
coreutils         # GNU core utilities
procps            # Process utilities (ps, top, etc.)
lsof              # List open files
htop              # Interactive process viewer
screen            # Terminal multiplexer
tmux              # Alternative terminal multiplexer

# ============================================
# OPTIONAL BUT RECOMMENDED
# ============================================
wireshark         # GUI packet analyzer (if running desktop)
termshark         # Terminal-based packet analyzer
grpcurl           # gRPC curl-like tool
httpie            # User-friendly HTTP client

# ============================================
# NOTES
# ============================================
# Package names may vary between distributions:
# - RHEL/Fedora/CentOS: bind-utils, nmap-ncat, iproute
# - Ubuntu/Debian: dnsutils, netcat-openbsd, iproute2
#
# The setup.sh script handles these differences automatically.
